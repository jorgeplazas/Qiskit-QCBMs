\begin{frame}{Scope: background }

	\begin{block}{Quantum Circuit Born Machines}
		\vskip 0.4 truecm
QCBMs are generative models that use quantum circuits to represent probability distributions. 
	\end{block}
		\vskip 0.4 truecm
\begin{center}
	Distribution of data $\rightleftarrows$ quantum state
\end{center}

\begin{eqnarray*}
\pi(X) &\sim&  |\braket{X | \psi_{\theta}}|^2
\end{eqnarray*}

%	\includegraphics[width=.5\textwidth]{Quantum_Computer.jpg}
	
\end{frame}

\begin{frame}{Scope: goal}
	
	\begin{block}{Project \# 64}
		\vskip 0.2 truecm
	This project aims at developing a set of tools for implementing Quantum Circuit Born Machines using Qiskit. 
	\end{block}
	
\end{frame}

\begin{frame}{Deliverables}
	
	\begin{block}{Minimal Viable Product (MVP):}
		\vskip 0.2 truecm
		A jupyter notebook serving as a primer on QCBMs with examples of their implementation within the framework of Qiskit patterns.
	\end{block}
	
\end{frame}

\begin{frame}{Deliverables}
	
	\begin{block}{Primary Deliverables}
		\vskip 0.2 truecm
	\begin{itemize}
		\item  A python module based on Qiskit for the implementation of QCBMs.  \pause
		
		\item A set of jupyter notebooks  implementing use cases of QCBMs within the qiskit patterns framework. \pause
